// Channels
type Channel = 'linkedin' | 'instagram' | 'whatsapp' | 'email' | 'phone' | 'cnpj';

// Lead with channels
interface Lead {
  id: string;
  name: string;
  title: string;
  company: string;
  email?: string;
  phone?: string;
  linkedinUrl?: string;
  instagramUsername?: string;
  channels: Channel[];           // â† CANAIS DISPONIVEIS
  icpScore: number;              // â† ICP SCORE
  cadenceStatus?: {              // â† STATUS DA CADENCIA
    cadenceId: string;
    currentDay: number;
    totalDays: number;
    nextActivity: Channel;
    nextActivityTime: Date;
  };
  status: 'available' | 'in_cadence' | 'responding' | 'scheduled' | 'converted';
}

// Cadence
interface Cadence {
  id: string;
  code: string;                  // ex: "OUT-HT-14"
  name: string;
  type: 'outbound' | 'inbound';
  touchLevel: 'low' | 'medium' | 'high';
  durationDays: number;
  totalActivities: number;
  channels: Channel[];
  activities: CadenceActivity[];
}

interface CadenceActivity {
  id: string;
  day: number;
  channel: Channel;
  type: 'research' | 'connect' | 'message' | 'call' | 'email' | 'voice_ai';
  timeSlot: 'morning' | 'afternoon';
  template?: string;
}

// Campaign with cadence
interface Campaign {
  id: string;
  name: string;
  type: 'connection' | 'warmup' | 'authority' | 'relationship' | 'instagram_dm' | 'multichannel';
  channels: Channel[];
  cadenceId: string;
  status: 'draft' | 'scheduled' | 'active' | 'paused' | 'completed';
  leadsCount: number;
  responsesCount: number;
  conversionRate: number;
}

// Pipeline Deal with Show-Rate
interface Deal {
  id: string;
  leadId: string;
  name: string;
  title: string;
  company: string;
  value: number;
  channels: Channel[];
  cadenceStatus?: CadenceStatus;
  stage: 'new' | 'relationship' | 'scheduled' | 'won' | 'lost';
  meeting?: {
    date: Date;
    showRateGuard: ShowRateStatus;
  };
}

interface ShowRateStatus {
  emailL24h: 'pending' | 'sent' | 'confirmed';
  whatsappL12h: 'pending' | 'sent' | 'confirmed';
  smsL2h: 'pending' | 'sent' | 'confirmed';
  whatsappL30m: 'pending' | 'sent' | 'confirmed';
}

// Conversation (multi-channel)
interface Conversation {
  id: string;
  leadId: string;
  leadName: string;
  company: string;
  channels: Channel[];           // Canais disponiveis para esse lead
  lastMessage: Message;
  unreadCount: number;
  cadenceStatus?: CadenceStatus;
}

interface Message {
  id: string;
  channel: Channel;             // â† DE QUAL CANAL VEIO
  content: string;
  direction: 'inbound' | 'outbound';
  isAiGenerated: boolean;
  sentAt: Date;
  status: 'sent' | 'delivered' | 'read' | 'scheduled' | 'failed';
}

// AI Agent
interface AIAgent {
  id: string;
  name: string;
  type: 'connection' | 'inbox' | 'comment' | 'content' | 'qualifier' | 'voice_ai';
  prompt: string;
  model: 'claude-sonnet' | 'claude-haiku' | 'claude-opus' | 'gpt-4';
  temperature: number;
  language: string;
}
